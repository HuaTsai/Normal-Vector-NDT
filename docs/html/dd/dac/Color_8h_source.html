7 Color.h
8 Random class for encapsulating colors...
9 
10 Due to all the possible colorspace variants, here's the documentation of
11 what is actually implemented for the conversions:
12  - CIE 1931 2-degree observer
13  - ITU-R BT.709 primaries
14  - D65 illuminant (5nm tabulated - not the rounded version in Rec. 709)
15  - CIELAB uses a linear adaptation from D65 to equal-intensity white so that
16  1,1,1 in RGB maps to 100,0,0 in CIELAB
17  - "RGB" means linearly-scaled RGB with the above illuminant and primaries
18  - RGB-sRGB conversion uses the full part-linear, part-power-law function
19  - HSV is single-hexcone, sRGB
20  - Y'CbCr is JFIF-standard (Rec. 601 scaling, full excursion) starting from sRGB
21  - Range of [0..1] for all spaces except CIELAB and hue in HSV
22 */

111  inline const Color hsv2srgb() const
112  {
113  using namespace ::std;
114 
115  // From FvDFH
116  float H = v[0], S = v[1], V = v[2];
117  if (S <= 0.0f)
118  return Color(V,V,V);
119  H = fmod(H, M_2PIf);
120  if (H < 0.0f)
121  H += M_2PIf;
122  H /= M_PI_3f;
123  int i = int(floor(H));
124  float f = H - i;
125  float P = V * (1.0f - S);
126  float Q = V * (1.0f - (S * f));
127  float T = V * (1.0f - (S * (1.0f - f)));
128  switch(i) {
129  case 0: return Color(V, T, P);
130  case 1: return Color(Q, V, P);
131  case 2: return Color(P, V, T);
132  case 3: return Color(P, Q, V);
133  case 4: return Color(T, P, V);
134  default: return Color(V, P, Q);
135  }

137  inline const Color srgb2hsv() const
138  {
139  using ::std::min;
140  using ::std::max;
141 
142  float V = max(max(v[0], v[1]), v[2]);
143  float diff = V - min(min(v[0], v[1]), v[2]);
144  float S = diff / V;
145  float H = 0.0f;
146  if (S == 0.0f)
147  return Color(H, S, V);
148  if (V == v[0])
149  H = (v[1] - v[2]) / diff;
150  else if (V == v[1])

152  else
