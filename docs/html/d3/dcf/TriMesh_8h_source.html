213  // Angle of corner j in triangle i
214  inline float cornerangle(int i, int j)
215  {
216  using namespace ::std;
217 
218  if (unlikely(faces.empty())) need_faces();
219  const point &p0 = vertices[faces[i][j]];
220  const point &p1 = vertices[faces[i][NEXT_MOD3(j)]];
221  const point &p2 = vertices[faces[i][PREV_MOD3(j)]];
222  return acos((p1 - p0) DOT (p2 - p0));
223  }

225  // Dihedral angle between face i and face across_edge[i][j]
226  inline float dihedral(int i, int j)
227  {
228  if (unlikely(across_edge.empty())) need_across_edge();
229  if (unlikely(across_edge[i][j] < 0)) return 0.0f;
230  vec mynorm = trinorm(i);
231  vec othernorm = trinorm(across_edge[i][j]);
232  float ang = angle(mynorm, othernorm);
233  vec towards = 0.5f * (vertices[faces[i][NEXT_MOD3(j)]] +
234  vertices[faces[i][PREV_MOD3(j)]]) -
235  vertices[faces[i][j]];
236  if ((towards DOT othernorm) < 0.0f)
237  return M_PIf + ang;
238  else
239  return M_PIf - ang;
240  }
