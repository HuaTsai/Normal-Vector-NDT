80  std::vector<Eigen::Matrix3d> Generate(size_t sizes) {
81  Expects(is_radius_set_ && is_angle_set_);
82  std::vector<Eigen::Matrix3d> ret(sizes, Eigen::Matrix3d::Identity());
83  std::uniform_real_distribution<> radius_urd(radius_min_ * radius_min_,
84  radius_max_ * radius_max_);
85  std::uniform_real_distribution<> theta_urd(0, 2 * M_PI);
86  std::uniform_real_distribution<> angle_urd(angle_rad_min_, angle_rad_max_);
87  for (size_t i = 0; i < sizes; ++i) {
88  double radius = sqrt(radius_urd(*dre_));
89  double theta = theta_urd(*dre_);
90  double angle = angle_urd(*dre_);
91  ret.at(i)(0, 2) = center_x_ + radius * cos(theta);
92  ret.at(i)(1, 2) = center_y_ + radius * sin(theta);
93  ret.at(i)(0, 0) = cos(center_th_ + angle);
94  ret.at(i)(0, 1) = -sin(center_th_ + angle);
95  ret.at(i)(1, 0) = sin(center_th_ + angle);
96  ret.at(i)(1, 1) = cos(center_th_ + angle);
97  }
98  return ret;
99  }

340 std::vector<std::vector<double>> Combinations(std::vector<std::vector<double>> x) {
341  /* input: {{0, 1}, {a, b}, {x, y}}
342  * output: {{0, a, x}, {0, a, y}, ..., {1, b, y}}
343  */
344  std::vector<std::vector<double>> ret;
345  std::vector<std::vector<double>::iterator> its;
346  for (size_t i = 0; i < x.size(); ++i) {
347  its.push_back(x.at(i).begin());
348  }
349  while (its.at(0) != x.at(0).end()) {
350  std::vector<double> tmp;
351  for (size_t i = 0; i < x.size(); ++i) {
352  tmp.push_back(*its.at(i));
353  }
354  ret.push_back(tmp);
355  ++its.back();
356  for (size_t i = x.size() - 1; (i > 0) && (its.at(i) == x.at(i).end()); --i) {
357  its.at(i) = x.at(i).begin();
358  ++its.at(i - 1);
359  }
360  }
361  return ret;
362 }
