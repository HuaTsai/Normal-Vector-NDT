259 static inline unsigned xorshift_rnd(unsigned n = ~0u)
260 {

277 // If n > 0, returns values in [0, n).

294 // Return uniformly-distributed numbers. Version for floating-point types.
295 // If n > 0, returns values in (0, n).
296 // If n < 0, returns values in (-n , n).
297 template <class T>
298 static inline
299 typename ::std::enable_if< ::std::is_floating_point<T>::value, T >::type
300 uniform_rnd(T n)
301 {
302  if (n > 0) {
303  const T scale = T(1) / ::std::numeric_limits<unsigned>::max();
304  return T(xorshift_rnd()) * scale * n;
305  } else if (n < 0) {

321 // Returns normal-distributed (Gaussian) random numbers with the given sigma.
322 // Uses Marsaglia's polar form of the Box-Muller method.
323 template <class T>
324 static inline
325 typename ::std::enable_if< ::std::is_floating_point<T>::value, T >::type
326 normal_rnd(T sigma)
327 {
328  if (sigma == 0)
329  return 0;
330 
331  static bool have_saved = false;
332  static T saved = 0;
333  if (have_saved) {
334  have_saved = false;
335  return sigma * saved;
336  }
337 
338  T x, y, r2;
339  do {
340  x = uniform_rnd(T(-1));
341  y = uniform_rnd(T(-1));
342  r2 = sqr(x) + sqr(y);
343  } while (r2 >= T(1));
344 
345  if (r2 == T(0))
346  return T(0);
347 
348  T g = sqrt(T(-2) * log(r2) / r2);
349  saved = x * g;
350  have_saved = true;
351  return sigma * y * g;
352 }
