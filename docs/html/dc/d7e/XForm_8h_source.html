224  static inline XForm<T> rot_into(T d1x, T d1y, T d1z, T d2x, T d2y, T d2z)
225  {
226  using namespace ::std;
227 
228  // Normalize input directions
229  T l1 = sqrt(d1x*d1x + d1y*d1y + d1z*d1z);
230  T l2 = sqrt(d2x*d2x + d2y*d2y + d2z*d2z);
231  if (unlikely(l1 == 0 || l2 == 0))
232  return XForm<T>();
233  T rl1 = 1 / l1; d1x *= rl1; d1y *= rl1; d1z *= rl1;
234  T rl2 = 1 / l2; d2x *= rl2; d2y *= rl2; d2z *= rl2;
235 
236  // Find cosine of rotation angle
237  T c = d1x*d2x + d1y*d2y + d1z*d2z;
238  if (c <= -1) {
239  // Rotation of 180 degrees. Find an arbitrary
240  // rotation axis perpendicular to d1 and d2
241  T ax = d1y, ay = -d1x; // az = 0
242  if (ax == 0 && ay == 0)
243  ax = 1;
244  T rla = 1 / sqrt(ax*ax + ay*ay);
245  ax *= rla;
246  ay *= rla;
247 
248  // Construct rotation matrix
249  return XForm<T>(2*ax*ax-1, 2*ax*ay, 0, 0,
250  2*ax*ay, 2*ay*ay-1, 0, 0,
251  0, 0, -1, 0,
252  0, 0, 0, 1);
253  }
254 
255  // Else construct rotation axis * sin(angle)
256  T sx = d1y * d2z - d1z * d2y;
257  T sy = d1z * d2x - d1x * d2z;
258  T sz = d1x * d2y - d1y * d2x;
259  // We need to multiply by (1 - cos) / sin^2 = 1 / (1 + cos)
260  T r1c = 1 / (1 + c);
261  return XForm<T>(sx*sx*r1c+c, sx*sy*r1c+sz, sx*sz*r1c-sy, 0,
262  sx*sy*r1c-sz, sy*sy*r1c+c, sy*sz*r1c+sx, 0,
263  sx*sz*r1c+sy, sy*sz*r1c-sx, sz*sz*r1c+c, 0,
264  0, 0, 0, 1);
265  }

563 // Decompose the rotational part of an xform into angle and axis
564 template <class T, class S>
565 static inline void decompose_rot(const XForm<T> &xf, T &angle, S &axis)
566 {
567  using namespace ::std;
568 
569  // Initial estimate from antisymmetric component
570  T ax = xf[6] - xf[9];
571  T ay = xf[8] - xf[2];
572  T az = xf[1] - xf[4];
573  T l = sqrt(ax*ax + ay*ay + az*az);
574  T tr1 = xf[0] + xf[5] + xf[10] - 1.0;
575  angle = atan2(l, tr1);
576 
577  if (angle < T(M_PI_2l)) {
578  axis[0] = ax;
579  axis[1] = ay;
580  axis[2] = az;
581  } else {
582  // If rotation is over pi/2, using the symmetric component
583  // to find the rotation axis is more accurate than
584  // using the antisymmetric part.
585  // But we still need the latter to disambiguate flip.
586  T tx = abs(2 * xf[0] - tr1);
587  T ty = abs(2 * xf[5] - tr1);
588  T tz = abs(2 * xf[10] - tr1);
589  if (tx > ty && tx > tz) {
590  axis[0] = 2 * xf[0] - tr1;
591  axis[1] = xf[1] + xf[4];
592  axis[2] = xf[8] + xf[2];
593  if (xf[6] - xf[9] < 0) {
594  axis[0] = -axis[0];
595  axis[1] = -axis[1];
596  axis[2] = -axis[2];
597  }
598  } else if (ty > tx && ty > tz) {
599  axis[0] = xf[1] + xf[4];
600  axis[1] = 2 * xf[5] - tr1;
601  axis[2] = xf[6] + xf[9];
602  if (xf[8] - xf[2] < 0) {
603  axis[0] = -axis[0];
604  axis[1] = -axis[1];
605  axis[2] = -axis[2];
606  }
607  } else {
608  axis[0] = xf[8] + xf[2];
609  axis[1] = xf[6] + xf[9];
610  axis[2] = 2 * xf[10] - tr1;
611  if (xf[1] - xf[4] < 0) {
612  axis[0] = -axis[0];
613  axis[1] = -axis[1];
614  axis[2] = -axis[2];
615  }
616  }
617  }
618 
619  T len_ax = sqrt(sqr(axis[0]) + sqr(axis[1]) + sqr(axis[2]));
620  if (likely(len_ax != 0)) {
621  T scale = 1 / len_ax;
622  axis[0] *= scale;
623  axis[1] *= scale;
624  axis[2] *= scale;
625  } else {
626  axis[2] = 1;
627  }
628 }

631 // Make orthonormal
632 template <class T>
633 static inline void orthogonalize(XForm<T> &xf)
634 {
635  if (unlikely(xf[15] == 0)) { // Yuck. Doesn't make sense...
636  xf[15] = 1;
637  } else if (unlikely(xf[15] != 1)) {
638  T scale = 1 / xf[15];
639  for (size_t i = 0; i < 15; i++)
640  xf[i] *= scale;
641  xf[15] = 1;
642  }
643 
644  T angle;
645  Vec<3,T> axis;
646  decompose_rot(xf, angle, axis);
647  Vec<3,T> trans(xf[12], xf[13], xf[14]);
648 
649  xf = XForm<T>::rot(angle, axis);
650  xf[12] = trans[0];
651  xf[13] = trans[1];
652  xf[14] = trans[2];
653 }

656 // Interpolate between xforms, returning an xform that's a fraction "a" of
657 // the way from x to y. Performs screw decomposition, interpolates rotation
658 // and translation along screw axis, and constructs in-plane translation.
659 template <class T, class S>
660 static inline XForm<T> mix(const XForm<T> &x, const XForm<T> &y, const S &a)
661 {
662  XForm<T> x2y = y * inv(x);
663 
664  // Find angle/axis of rotation
665  T angle;
666  Vec<3,T> axis;
667  decompose_rot(x2y, angle, axis);
668 
669  // Split the translation into two components:
670  // along axis, and in the plane perpendicular to axis
671  Vec<3,T> trans(x2y[12], x2y[13], x2y[14]);
672  Vec<3,T> along = (trans DOT axis) * axis;
673  Vec<3,T> in_plane = trans - along;
674  Vec<3,T> perp = in_plane CROSS axis;
675 
676  // Now build the interpolated transform
677  T theta = T(0.5) * angle, a1 = 1 - a;
678  T factor = (theta > T(1.0e-8)) ? sin(a * theta) / sin(theta) : a;
679  Vec<3,T> trans_out = a * along +
680  factor * cos(a1 * theta) * in_plane +
681  factor * sin(a1 * theta) * perp;
682 
683  return XForm<T>::trans(trans_out) *
684  XForm<T>::rot(a * angle, axis) * x;
685 }
