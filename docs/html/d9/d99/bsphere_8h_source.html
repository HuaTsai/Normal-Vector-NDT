99 template <size_t D, class T>
100 bool Basis<D,T>::push(const Vec<D,T> &p)
101 {
102  const T eps = T(1.0e-13);
103  if (m == 0) {
104  for (size_t i = 0; i < D; i++)
105  c[0][i] = q0[i] = p[i];
106  sqr_r[0] = 0;
107  } else {
108  // set v_m to Q_m
109  for (size_t i = 0; i < D; i++)
110  v[m][i] = p[i] - q0[i];
111 
112  // compute the a_{m,i}, i < m
113  for (size_t i = 1; i < m; i++) {
114  a[m][i] = 0;
115  for (size_t j = 0; j < D; j++)
116  a[m][i] += v[i][j] * v[m][j];
117  a[m][i] *= (T(2) / z[i]);
118  }
119 
120  // update v_m to Q_m-\bar{Q}_m
121  for (size_t i = 1; i < m; i++) {
122  for (size_t j = 0; j < D; j++)
123  v[m][j] -= a[m][i] * v[i][j];
124  }
125 
126  // compute z_m
127  z[m] = 0;
128  for (size_t j = 0; j < D; j++)
129  z[m] += sqr(v[m][j]);
130  z[m] *= T(2);
131 
132  // reject push if z_m too small
133  if (z[m] < eps * current_sqr_r)
134  return false;
135 
136  // update c, sqr_r
137  T e = -sqr_r[m-1];
138  for (size_t i = 0; i < D; i++)
139  e += sqr(p[i] - c[m-1][i]);
140  f[m] = e / z[m];
141 
142  for (size_t i = 0; i < D; i++)
143  c[m][i] = c[m-1][i] + f[m]*v[m][i];
144  sqr_r[m] = sqr_r[m-1] + T(0.5) * e * f[m];
145  }
146  current_c = c[m];
147  current_sqr_r = sqr_r[m];
148  s = ++m;
149  return true;
150 }

170 T Miniball<D,T>::max_excess(It t, It i, It &pivot) const
171 {
172  const T *c = B.center(), sqr_r = B.squared_radius();
173  T e, max_e = 0;
174  for (It k = t; k != i; k++) {
175  const Vec<D,T> &p = *k;

178  e += sqr(p[j] - c[j]);

207 template <size_t D, class T>
208 void Miniball<D,T>::pivot_mb(It i)
209 {
210  It t = ++L.begin();
211  mtf_mb(t);
212  T max_e, old_sqr_r = T(0);
213  do {
214  It pivot;
215  max_e = max_excess(t, i, pivot);
216  if (max_e > 0) {
217  t = support_end;
218  if (t == pivot)
219  t++;
220  old_sqr_r = B.squared_radius();
221  B.push(*pivot);
222  mtf_mb(support_end);
223  B.pop();
224  move_to_front(pivot);
225  }
226  } while (max_e > 0 && B.squared_radius() > old_sqr_r);
227 }
