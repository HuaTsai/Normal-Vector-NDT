240  bool computePointNormal2d(boost::shared_ptr<std::vector<int>>& indices,
241  float& nx, float& ny, float& nz) const {
242  if (indices->size() < 2) {
243  nx = ny = nz = std::numeric_limits<float>::quiet_NaN();
244  return false;
245  }
246  if (indices->size() == 2) {
247  double norm, vect_x, vect_y;
248  vect_x = m_in_cloud->points[(*indices)[0]].x -
249  m_in_cloud->points[(*indices)[1]].x;
250  vect_y = m_in_cloud->points[(*indices)[0]].y -
251  m_in_cloud->points[(*indices)[1]].y;
252  norm = std::pow(std::pow(vect_x, 2.0) + std::pow(vect_y, 2.0), 0.5);
253  vect_x /= norm;
254  vect_y /= norm;
255  nx = -vect_y;
256  ny = vect_x;
257  if (ny < 0) {
258  nx = -nx;
259  ny = -ny;
260  }
261  nz = 0.0;
262  return true;
263  }
264 
265  // Get the plane normal and surface curvature
266  PCA2D pca;
267 
268  pca.setInputCloud(m_in_cloud);
269  pca.setIndices(indices);
270  auto result = pca.getEigenVectors().col(1);
271  nx = result(0);
